Here's everything you need to know about Dockerfiles and Docker Compose:

## **DOCKERFILE COMMANDS**

### Essential Instructions:
```dockerfile
# Base image
FROM node:18-alpine

# Set working directory
WORKDIR /app

# Copy files (source → dest)
COPY package.json ./
COPY . .

# Copy with ownership
COPY --chown=node:node . .

# Run commands during build
RUN npm install
RUN apt-get update && apt-get install -y curl

# Set environment variables
ENV NODE_ENV=production
ENV PORT=3000

# Expose ports (documentation)
EXPOSE 3000

# User to run container (security)
USER node

# Command to run (can be overridden)
CMD ["node", "app.js"]

# Entrypoint (harder to override)
ENTRYPOINT ["npm", "start"]

# Add metadata labels
LABEL version="1.0"
LABEL maintainer="email@example.com"

# Build arguments (set at build time)
ARG VERSION=latest

# Volume mount point
VOLUME /data

# Health check
HEALTHCHECK --interval=30s CMD curl -f http://localhost/ || exit 1
```

### Important Differences:
- **CMD** - Can be overridden by `docker run` command line arguments
- **ENTRYPOINT** - Cannot be overridden, only appended to
- **RUN** - Executes at build time (creates layers)
- **CMD/ENTRYPOINT** - Executes at container runtime

## **PULLING IMAGES EXAMPLES**

```bash
# Pull specific version

# Pull with digest (most specific)
```

## **DOCKERFILE EXAMPLES**

### Example 1: Python Flask App
```dockerfile
FROM python:3.11-slim

WORKDIR /app

# Install dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application
COPY . .

ENV FLASK_APP=app.py
ENV FLASK_ENV=production

EXPOSE 5000

USER nobody

CMD ["python", "-m", "flask", "run", "--host=0.0.0.0"]
```

### Example 2: Node.js API
```dockerfile
FROM node:18-alpine

WORKDIR /app

# Copy package files
COPY package*.json ./
RUN npm ci --only=production

# Copy source code
COPY src/ ./src

ENV NODE_ENV=production
EXPOSE 3000

# Non-root user
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nodejs -u 1001
USER nodejs

CMD ["node", "src/server.js"]
```

### Example 3: Multi-stage Build (optimized)
```dockerfile
# Build stage
FROM golang:1.21 AS builder
WORKDIR /app
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -o myapp .

# Runtime stage
FROM alpine:latest
RUN apk --no-cache add ca-certificates
WORKDIR /root/
COPY --from=builder /app/myapp .
EXPOSE 8080
CMD ["./myapp"]
```

## **DOCKER COMPOSE BASICS**

### Basic `docker-compose.yml` structure:
```yaml
version: '3.8'

services:
  web:
    build: .
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
    volumes:
      - ./data:/app/data
    depends_on:
      - db
  
  db:
    image: postgres:15
    environment:
      POSTGRES_PASSWORD: secret
    volumes:
      - postgres_data:/var/lib/postgresql/data

volumes:
  postgres_data:
```

### Complete Docker Compose Example:
```yaml
version: '3.8'

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile
      args:
        - VERSION=1.0
    container_name: myapp
    restart: unless-stopped
    ports:
      - "8080:8080"
      - "9090:9090"
    environment:
      - DB_HOST=database
      - REDIS_HOST=cache
    env_file:
      - .env.production
    volumes:
      - ./uploads:/app/uploads
      - static_data:/app/static
    depends_on:
      - database
      - redis
    networks:
      - app-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  database:
    image: postgres:15-alpine
    container_name: postgres
    restart: always
    environment:
      POSTGRES_DB: mydb
      POSTGRES_USER: admin
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    networks:
      - app-network

  redis:
    image: redis:7-alpine
    container_name: redis
    restart: always
    command: redis-server --appendonly yes
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - app-network

  nginx:
    image: nginx:alpine
    container_name: nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
    depends_on:
      - app
    networks:
      - app-network

networks:
  app-network:
    driver: bridge

volumes:
  postgres_data:
  redis_data:
  static_data:
```

### Useful Docker Compose Commands:
```bash
# Start services
docker-compose up -d

# Stop services
docker-compose down

# Rebuild and start
docker-compose up --build

# View logs
docker-compose logs -f app

# Run one-off commands
docker-compose exec app npm run migrate

# Stop and remove everything (including volumes)
docker-compose down -v
```

## **BEST PRACTICES SUMMARY**

1. **Use specific tags** (not `latest`) for base images
2. **Order layers** from least to most frequently changing
3. **Combine RUN commands** to reduce layers:
   ```dockerfile
   RUN apt-get update && apt-get install -y curl && rm -rf /var/lib/apt/lists/*
   ```
4. **Use multi-stage builds** for smaller final images
5. **Run as non-root user** for security
6. **Use `.dockerignore`** to exclude unnecessary files
7. **Keep secrets out of Dockerfiles** - use build args or environment files
8. **Set resource limits** in Docker Compose:
   ```yaml
   services:
     app:
       deploy:
         resources:
           limits:
             cpus: '0.5'
             memory: 512M
   ```

Start with simple Dockerfiles and gradually add complexity as you learn!
